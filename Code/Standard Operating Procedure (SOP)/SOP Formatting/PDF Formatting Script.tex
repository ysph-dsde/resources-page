% Standard Operating Procedure (SOP) LaTeX Formatting

% This LaTeX script formats Quarto-generated PDF handouts for DSDE SOPs and
% workshops. It centralizes all PDF styling so documents render consistently
% across authors and over time.

% - Brand typography and section styling (YaleNew/Barlow + DSDE palette)
% - Title page layout and page backgrounds (first page vs subsequent pages)
% - Code blocks with consistent indentation and highlighting (Verbatim-based)
% - Indented callout blocks compatible with Quarto callouts, including a
%   workshop-only custom “Discussion” callout type
%
% Usage:
% Reference this file from the Quarto YAML header (include-in-header or
% template inclusion) and ensure path variables (e.g., \rootdir, \stylesdir)
% match the document’s directory depth and repository layout.
%

  \AddToShipoutPictureBG*{%
    \put(0,0){%
      \includegraphics[width=\paperwidth,height=\paperheight]{\rootdir/SOP Title Page.png}%
    }%
  }%
}

  \AddToShipoutPictureBG*{%

% ------------------------------
% Custom title page layout
% ------------------------------
\makeatletter
\def\@maketitle{%
  \newpage
  \null
  \begin{center}%
    {\customtitlefont\@title\par}%
    \ifx\@subtitle\@empty\else%
      \vskip 2em%
      {\customsubtitlefont\@subtitle\par}%
    \fi%
    \ifx\@author\@empty\else%
      \vskip 1em%
      {\customauthorfont\@author\par}%
    \fi%
    \ifx\b@date\@empty\else%
      \vskip 1em%
      {\customdatefont\@date\par}%
    \fi%
  \end{center}%
  \vskip 1em%
}
\makeatother

% =========================================================
% Callout Blocks for Indenting
% =========================================================

% These tcolorbox “callout” environments recreate Quarto’s native callout blocks
% (Note/Warning/Important/Tip/Caution/Discussion) but with consistent indentation.
% Quarto callouts do not otherwise indent. These wrappers provide an indented, 
% framed block with a colored left rule, a light header band, and an icon + 
% optional title argument.
%
% The "Discussion" callout is a workshop-only custom callout type. Use this
% formatting available for both indented and non-indented callout-discussion
% usage.

% ------------------------------
% Callout Note
% ------------------------------
\definecolor{NoteBordercolor}{HTML}{4581EC}  % Frame + left rule color
\definecolor{NoteHeadercolor}{HTML}{F0F5FD}  % Title/header background band
\definecolor{NoteIconcolor}{HTML}{0858E5}    % Icon (and title accent) color

\newtcolorbox{callout-note}[1][]{%
  colframe=NoteBordercolor, % Outer frame color
  leftrule=0.8mm,           % Emphasized left rule thickness
  boxrule=0.2mm,            % Thin border on other sides
  colback=white,            % Body background
  colbacktitle=NoteHeadercolor, % Header background
  coltitle=black,           % Header text color
  title={\textcolor{NoteIconcolor}{\faInfo}\ \  #1}, % Icon + optional title text
  titlerule=0pt,            % No line separating title and body
  toptitle=1mm,             % Padding above title text
  bottomtitle=1mm,          % Padding below title text
  left=2mm, right=2mm,      % Inner left/right padding
  before skip=10pt,         % Space before the box
  after skip=10pt           % Space after the box
}
